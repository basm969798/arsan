STEP 1: Execute the following bash script to strictly create DTO skeletons and setup validation:

```bash
cd backend

# 1. Install necessary validation packages

# 2. Create DTO directories
mkdir -p src/modules/auth/api/dtos

# 3. Create Login DTO

import { IsEmail, IsString, MinLength } from 'class-validator';

export class LoginDto {
  @IsEmail()
  email: string;

  @IsString()
  @MinLength(6)
  password: string;
}
EOF

# 4. Create CreateOrder DTO (Fixed imports)
cat << 'EOF' > src/modules/orders/api/dtos/create-order.dto.ts
import { IsArray, IsOptional, IsString } from 'class-validator';

export class CreateOrderDto {
  @IsArray()
  items: any[];

  @IsOptional()
  @IsString()
  notes?: string;
}
EOF

# 5. Create CreatePart DTO for Catalog
cat << 'EOF' > src/modules/catalog/api/dtos/create-part.dto.ts
import { IsString, IsNumber, IsOptional } from 'class-validator';

export class CreatePartDto {
  @IsString()
  name: string;

  @IsString()
  oemNumber: string;

  @IsOptional()
  @IsNumber()
  basePrice?: number;
}
EOF

# 6. Update main.ts to include Global Validation Pipe
cat << 'EOF' > src/main.ts
import { NestFactory } from '@nestjs/core';
import { AppModule } from './app.module';
import { ValidationPipe } from '@nestjs/common';
import { GlobalExceptionFilter } from './common/filters/global-exception.filter';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // Activate Global Validation
  app.useGlobalPipes(new ValidationPipe({
    whitelist: true,
    forbidNonWhitelisted: true,
    transform: true,
  }));

  app.useGlobalFilters(new GlobalExceptionFilter());
  app.enableCors();

  const port = process.env.PORT || 8080;
  await app.listen(port, '0.0.0.0');

  console.log(`🚀 Arsan API Secured & Running on port ${port}`);
}
bootstrap();
EOF
