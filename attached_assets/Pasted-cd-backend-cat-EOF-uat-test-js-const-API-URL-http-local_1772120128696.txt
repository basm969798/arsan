const API_URL = 'http://localhost:8080';
let token = '';

async function apiCall(endpoint, method = 'GET', body = null) {
  const headers = { 'Content-Type': 'application/json' };
  if (token) headers['Authorization'] = `Bearer ${token}`;
  
  const options = { method, headers };
  if (body) options.body = JSON.stringify(body);
  
  const res = await fetch(API_URL + endpoint, options);
  const data = await res.json().catch(() => ({}));
  
  if (!res.ok) throw new Error(`فشل في ${endpoint}: ${res.status} - ${JSON.stringify(data)}`);
  return data;
}

      email: email,
      password: 'password123'
    });
    const loginRes = await apiCall('/auth/login', 'POST', { email, password: 'password123' });
    token = loginRes.access_token || loginRes.token; // دعم للنسختين
    if (!token) throw new Error('لم يتم استلام Token بعد تسجيل الدخول!');
    console.log('✅ تم التسجيل والدخول بنجاح (تم استلام Token).');

    console.log('\nالخطوة 2: 📁 إنشاء قسم في الكتالوج');
    const category = await apiCall('/catalog/categories', 'POST', {
      name: 'فلاتر ومحركات',
      description: 'قسم مخصص لاختبار UAT'
    });
    categoryId = category.id || category.categoryId; // حسب استجابة الباك اند
    // إذا لم يرجع الـ POST الآي دي مباشرة، نجلبه من الـ GET
    if (!categoryId) {
      const cats = await apiCall('/catalog');
      categoryId = cats[0].id;
    }
    console.log('✅ تم إنشاء القسم بنجاح.');

    // ---------------------------------------------------------
    console.log('\nالخطوة 3: ⚙️ إضافة 3 قطع غيار للكتالوج');
    const parts = [
      { categoryId, name: 'فلتر زيت تويوتا', oemNumber: 'OEM-1001', basePrice: 45 },
      { categoryId, name: 'بواجي نيسان', oemNumber: 'OEM-1002', basePrice: 120 },
      { categoryId, name: 'فحمات فرامل كيا', oemNumber: 'OEM-1003', basePrice: 250 }
    ];
    for (const p of parts) {
      await apiCall('/catalog/parts', 'POST', p);
    }
    console.log('✅ تم حفظ 3 قطع غيار في قاعدة البيانات.');

    // ---------------------------------------------------------
    console.log('\nالخطوة 4: 🔍 البحث الذكي عن القطعة');
    const searchRes = await apiCall('/search/parts?query=OEM-1002');
    if (searchRes.length === 0) throw new Error('فشل محرك البحث في إيجاد القطعة!');
    partId = searchRes[0].id;
    console.log(`✅ محرك البحث يعمل! تم العثور على: ${searchRes[0].name} بسعر ${searchRes[0].basePrice} ر.س`);

    // ---------------------------------------------------------
    console.log('\nالخطوة 5: 🛒 إنشاء طلب جديد');
    const orderRes = await apiCall('/orders', 'POST', {
      items: [{ partId: partId, quantity: 2 }],
      notes: 'طلب عاجل من اختبار UAT'
    });
    
    // جلب قائمة الطلبات للتأكد من وجوده
    const allOrders = await apiCall('/orders');
    if (allOrders.length === 0) throw new Error('لم يتم تسجيل الطلب في قائمة الطلبات!');
    orderId = allOrders[allOrders.length - 1].id;
    console.log(`✅ تم إنشاء الطلب بنجاح. رقم الطلب: ${orderId} | الحالة: ${allOrders[allOrders.length - 1].status}`);

    // ---------------------------------------------------------
    console.log('\nالخطوة 6: 💰 الإغلاق المالي (السداد)');
    await apiCall('/finance/close-order', 'POST', {
      orderId: orderId,
      type: 'CASH',
      amount: 240, // 120 * 2
      note: 'تم السداد نقداً عبر اختبار UAT'
    });
    
    // التحقق من حالة الطلب بعد السداد
    const ordersAfterPay = await apiCall('/orders');
    const paidOrder = ordersAfterPay.find(o => o.id === orderId);
    console.log(`✅ تمت عملية السداد. الحالة الحالية للطلب: ${paidOrder.status}`);

    // ---------------------------------------------------------
    console.log('\n🏆 [النتيجة النهائية]: نجاح تام بنسبة 100%!');
    console.log('تم اختبار دورة B2B الكاملة (تسجيل -> كتالوج -> بحث -> طلب -> سداد) بدون أي أخطاء.');

  } catch (error) {
    console.error('\n❌ [فشل في الاختبار]:', error.message);
  }
}
