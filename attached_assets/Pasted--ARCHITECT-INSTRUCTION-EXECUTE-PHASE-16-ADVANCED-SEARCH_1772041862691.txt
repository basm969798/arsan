import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, Like } from 'typeorm';
import { Part } from '../../../catalog/domain/entities/part.entity';

@Injectable()
export class SearchService {
  constructor(
    @InjectRepository(Part)
    private readonly partRepo: Repository<Part>,
  ) {}

  async searchParts(query: string): Promise<Part[]> {
    console.log(`[SearchService] 🔍 Executing advanced search for: ${query}`);
    
    // Search by name or OEM Number
    return this.partRepo.find({
      where: [
        { name: Like(`%${query}%`) },
        { oemNumber: Like(`%${query}%`) }
      ],
      relations: ['category'] 
    });
  }

  async indexPart(partData: any): Promise<void> {
    // This is called by Sagas to update search indexes (Event-driven)
    console.log(`[SearchService] 🗂️ Indexing update received for: ${partData.id}`);
  }
}
EOF

import { Controller, Get, Query, UseGuards } from '@nestjs/common';
import { SearchService } from '../../application/services/search.service';
import { SearchQueryDto } from '../dtos/search-query.dto';
import { JwtAuthGuard } from '../../../auth/infrastructure/guards/jwt-auth.guard';

@UseGuards(JwtAuthGuard)
@Controller('search')
export class SearchController {
  constructor(private readonly searchService: SearchService) {}

  @Get('parts')
  async search(@Query() dto: SearchQueryDto) {
    return this.searchService.searchParts(dto.query);
  }
}
